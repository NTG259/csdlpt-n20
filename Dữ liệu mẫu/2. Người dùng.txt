-- =========================================================
-- SINH 1000 BAN GHI NGUOI DUNG CHO 2 SITE BAC / NAM
-- 500 BAN GHI BAC, 500 BAN GHI NAM
-- =========================================================

SET NOCOUNT ON;

BEGIN TRY
    BEGIN TRANSACTION;

    -- 1. Kiểm tra site Bac và Nam đã tồn tại trong bảng KhuVuc chưa
    IF NOT EXISTS (SELECT 1 FROM KhuVuc WHERE MaKhuVuc = 'Bac')
    BEGIN
        THROW 50001, N'Site Bac chưa tồn tại trong bảng KhuVuc.', 1;
    END;

    IF NOT EXISTS (SELECT 1 FROM KhuVuc WHERE MaKhuVuc = 'Nam')
    BEGIN
        THROW 50002, N'Site Nam chưa tồn tại trong bảng KhuVuc.', 1;
    END;

    -- 2. Tạo bảng tạm để lưu danh sách người dùng cần sinh
    IF OBJECT_ID('tempdb..#NguoiDungSeed') IS NOT NULL
        DROP TABLE #NguoiDungSeed;

    CREATE TABLE #NguoiDungSeed
    (
        STT INT NOT NULL,
        MaND UNIQUEIDENTIFIER NOT NULL,
        MatKhau VARCHAR(255) NOT NULL,
        MaKV VARCHAR(10) NOT NULL,
        MaKhoPhuTrach VARCHAR(20) NULL,
        HoTen NVARCHAR(100) NOT NULL,
        Email VARCHAR(100) NOT NULL,
        SoDienThoai VARCHAR(15) NOT NULL,
        DiaChi NVARCHAR(300) NULL,
        NgayDangKy DATE NOT NULL,
        TrangThai TINYINT NOT NULL,
        NgaySinh DATETIME2 NULL,
        GioiTinh NVARCHAR(10) NOT NULL,
        CCCD VARCHAR(12) NULL,
        VaiTro VARCHAR(20) NOT NULL
    );

    -- 3. Sinh 1000 dòng dữ liệu mẫu
    ;WITH N AS
    (
        SELECT TOP (1000)
            ROW_NUMBER() OVER (ORDER BY (SELECT NULL)) AS STT
        FROM sys.all_objects a
        CROSS JOIN sys.all_objects b
    )
    INSERT INTO #NguoiDungSeed
    (
        STT,
        MaND,
        MatKhau,
        MaKV,
        MaKhoPhuTrach,
        HoTen,
        Email,
        SoDienThoai,
        DiaChi,
        NgayDangKy,
        TrangThai,
        NgaySinh,
        GioiTinh,
        CCCD,
        VaiTro
    )
    SELECT
        STT,
        NEWID() AS MaND,

        -- Mật khẩu mẫu
        '123456' AS MatKhau,

        -- 500 người đầu thuộc Bac, 500 người sau thuộc Nam
        CASE 
            WHEN STT <= 500 THEN 'Bac'
            ELSE 'Nam'
        END AS MaKV,

        -- Để NULL để không phụ thuộc vào mã kho cụ thể
        NULL AS MaKhoPhuTrach,

        -- Họ tên mẫu
        CASE 
            WHEN STT <= 500 
                THEN CONCAT(N'Người dùng Bắc ', STT)
            ELSE 
                CONCAT(N'Người dùng Nam ', STT - 500)
        END AS HoTen,

        -- Email không trùng
        CASE 
            WHEN STT <= 500 
                THEN CONCAT('user_bac_', RIGHT('0000' + CAST(STT AS VARCHAR(4)), 4), '@gmail.com')
            ELSE 
                CONCAT('user_nam_', RIGHT('0000' + CAST(STT - 500 AS VARCHAR(4)), 4), '@gmail.com')
        END AS Email,

        -- Số điện thoại không trùng
        CONCAT('09', RIGHT('00000000' + CAST(STT AS VARCHAR(8)), 8)) AS SoDienThoai,

        -- Địa chỉ mẫu
        CASE 
            WHEN STT <= 500 
                THEN CONCAT(N'Địa chỉ mẫu tại site Bắc số ', STT)
            ELSE 
                CONCAT(N'Địa chỉ mẫu tại site Nam số ', STT - 500)
        END AS DiaChi,

        -- Ngày đăng ký rải trong 365 ngày gần đây
        DATEADD(DAY, - (STT % 365), CONVERT(DATE, GETDATE())) AS NgayDangKy,

        -- Trạng thái hoạt động
        1 AS TrangThai,

        -- Ngày sinh mẫu, tuổi khoảng 18-47
        DATEADD(YEAR, - (18 + (STT % 30)), GETDATE()) AS NgaySinh,

        -- Giới tính xen kẽ
        CASE 
            WHEN STT % 2 = 0 THEN N'Nam'
            ELSE N'Nữ'
        END AS GioiTinh,

        -- CCCD 12 ký tự
        RIGHT('000000000000' + CAST(100000000000 + STT AS VARCHAR(12)), 12) AS CCCD,

        -- Vai trò: đa số là USER
        'USER' AS VaiTro
    FROM N;

    -- 4. Kiểm tra trùng Email hoặc SĐT trước khi insert
    IF EXISTS
    (
        SELECT 1
        FROM #NguoiDungSeed s
        JOIN NguoiDung nd ON nd.Email = s.Email
    )
    BEGIN
        THROW 50003, N'Có Email đã tồn tại trong bảng NguoiDung. Hãy đổi mẫu email trước khi chạy lại.', 1;
    END;

    IF EXISTS
    (
        SELECT 1
        FROM #NguoiDungSeed s
        JOIN NguoiDung nd ON nd.SoDienThoai = s.SoDienThoai
    )
    BEGIN
        THROW 50004, N'Có số điện thoại đã tồn tại trong bảng NguoiDung. Hãy đổi mẫu số điện thoại trước khi chạy lại.', 1;
    END;

    IF EXISTS
    (
        SELECT 1
        FROM #NguoiDungSeed s
        JOIN User_Global_Index ugi ON ugi.Email = s.Email
    )
    BEGIN
        THROW 50005, N'Có Email đã tồn tại trong bảng User_Global_Index. Hãy đổi mẫu email trước khi chạy lại.', 1;
    END;

    IF EXISTS
    (
        SELECT 1
        FROM #NguoiDungSeed s
        JOIN User_Global_Index ugi ON ugi.SoDienThoai = s.SoDienThoai
    )
    BEGIN
        THROW 50006, N'Có số điện thoại đã tồn tại trong bảng User_Global_Index. Hãy đổi mẫu số điện thoại trước khi chạy lại.', 1;
    END;

    -- 5. Insert vào bảng NguoiDung
    INSERT INTO NguoiDung
    (
        MaND,
        MatKhau,
        MaKV,
        MaKhoPhuTrach,
        HoTen,
        Email,
        SoDienThoai,
        DiaChi,
        NgayDangKy,
        TrangThai,
        NgaySinh,
        GioiTinh,
        CCCD,
        VaiTro
    )
    SELECT
        MaND,
        MatKhau,
        MaKV,
        MaKhoPhuTrach,
        HoTen,
        Email,
        SoDienThoai,
        DiaChi,
        NgayDangKy,
        TrangThai,
        NgaySinh,
        GioiTinh,
        CCCD,
        VaiTro
    FROM #NguoiDungSeed;

    -- 6. Insert vào bảng User_Global_Index
    INSERT INTO User_Global_Index
    (
        MaND,
        Email,
        SoDienThoai,
        MaKhuVuc
    )
    SELECT
        MaND,
        Email,
        SoDienThoai,
        MaKV
    FROM #NguoiDungSeed;

    COMMIT TRANSACTION;

    PRINT N'Đã sinh thành công 1000 bản ghi NguoiDung và 1000 bản ghi User_Global_Index.';

END TRY
BEGIN CATCH
    IF @@TRANCOUNT > 0
        ROLLBACK TRANSACTION;

    DECLARE @ErrMsg NVARCHAR(4000) = ERROR_MESSAGE();
    DECLARE @ErrSeverity INT = ERROR_SEVERITY();
    DECLARE @ErrState INT = ERROR_STATE();

    RAISERROR(@ErrMsg, @ErrSeverity, @ErrState);
END CATCH;
GO
